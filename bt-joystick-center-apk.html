Servo servo_X;

int Z_axis = 2;

int x = 0;

int y = 0;

int z = 0;

int pos = 0;

int check1 = 0;

int check2 = 0;

void setup()

{

Serial.begin(9600);

servo_X.attach(7);

servo_Y.attach(6);

pinMode(X_axis, INPUT);

pinMode(Y_axis, INPUT);

pinMode(Z_axis, INPUT);

digitalWrite(Z_axis, HIGH);

}

void loop()

{

x = analogRead(X_axis);

y = analogRead(Y_axis);

z = digitalRead(Z_axis);

if(z == LOW)

{

Serial.print("Z axis status = ");

Serial.println("Button Pressed");

Serial.println("Sweeping servo actuators");

for (pos = 0; pos <= 180; pos += 1)

{

servo_X.write(pos);

delay(10);

}

for (pos = 180; pos >= 0; pos -= 1)

{

servo_X.write(pos);

delay(15);

}

for (pos = 0; pos <= 180; pos += 1)

{

servo_Y.write(pos);

delay(10);

}

for (pos = 180; pos >= 0; pos -= 1)

{

servo_Y.write(pos);

delay(15);

}

Serial.println("Done!!!");

}

if(x > check1 + threshold || x < check1 - threshold)

{

X_angleValue = map(x, 0, 1023, 0, 180);

Serial.println(" degree");

Serial.println("------------------------------------------");

}

if(y > check2 + threshold || y < check2 - threshold)

30. Sebutkan 4 kekurangan dari Joystick ​


Beberapa kekurangan joystick antara lain:

Delay. Delay adalah kondisi saat kita menekan tombol yang ada di joystick terdapat jeda nol koma sekian detik sebelum perintah input dari joystick dapat terbaca di komputer/ HP. Hal ini sering terjadi pada joystick wireless yang dipengaruhi oleh kekuatan bluetooth atau wifi yang menjadi perantara.Tidak semua device mendudukung adanya perangkat tambahan joystick. Beberapa jenis joystick ada yang membutuhkan spesifikasi device tertentu. Misal pada joystick wired yang membutuhkan HP yang support koneksi OTG.Analog drifting. Drifting terjadi saat kondisi analog seharusnya diam ketika tidak kita sentuh atau gerakkan tapi pada kondisi drifting ini analog dapat bergerak dengan sendiri secara random.Sedikit ribet, baik saat instalasi driver jika joystick bukan jenis plug-n-play dan juga saat setting joystick sebelum bermain game. Namun ini tidak begitu bermasalah karena biasanya ada guide dan tutorial instalasi serta setting-nya.
