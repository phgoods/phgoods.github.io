[tex] \sf \displaystyle\sum_{a=2}^{4} \: ( \sf2a + 5) \\ \\ \sf n = 4 - 2 + 1( + 1 \: karena \: batas \: bawah \: tidak \: dimulai \: dari \: 1) \\ \sf n = 3 \\ \\ \sf a_{1} = 2 \times 2 + 5 = 9 \\ \sf a_{3} = 2 \times4 + 5 = 13 \\ \\ \sf S_{n} = n \times \frac{a_{1} + a_{3}}{2} \\ \sf S_{3} = 3 \times \frac{9 + 13}{2} \\ \sf S_{3} = 3 \times \frac{22}{2} \\ \sf S_{3} = 3 \times 11 \\ \\ \boxed{ \sf S_{3} = 33}[/tex]

[tex] = \displaystyle\sum_{a=2}^{4} \: (2a + 5)[/tex]

[tex] = \displaystyle\sum_{a=2}^{4} \: 2a + \displaystyle\sum_{a=2}^{4} \: 5[/tex]
